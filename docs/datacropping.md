 * Pseudo-code for Temporal Alignment
 * Ensures that the 'index' of the glove array matches the 'index' of the vision array
 */
function cropMultimodalSession(sessionData, startFrame, endFrame) {
    const croppedGlove = sessionData.gloveStream.slice(startFrame, endFrame);
    const croppedVision = sessionData.visionStream.slice(startFrame, endFrame);
    
    // Integrity Check: Ensure 1:1 Mapping
    if (croppedGlove.length !== croppedVision.length) {
        throw new Error("Alignment Desync: Glove and Vision sample counts do not match.");
    }
    
    return { croppedGlove, croppedVision };
}
